name: Build Debug APK

on:
  workflow_dispatch:

jobs:
  build:
    runs-on: ubuntu-latest
    permissions:
      contents: read

    steps:
      - name: Checkout repository
        uses: actions/checkout@v4
        
      - name: Set up JDK 21
        uses: actions/setup-java@v4
        with:
          java-version: '21'
          distribution: 'temurin'

      - name: Create local.properties
        run: echo "sdk.dir=$ANDROID_SDK_ROOT" > catchit-android/local.properties

      # Remove daemon JVM properties so Gradle uses the system JDK (JDK 21 from setup-java).
      # The file points to foojay.io for JVM downloads which adds latency and a network dependency.
      - name: Remove gradle-daemon-jvm.properties
        run: rm -f catchit-android/gradle/gradle-daemon-jvm.properties

      - name: Cache Gradle packages
        uses: actions/cache@v4
        with:
          path: |
            ~/.gradle/caches
            ~/.gradle/wrapper
          key: ${{ runner.os }}-gradle-${{ hashFiles('catchit-android/**/*.gradle.kts', 'catchit-android/gradle/wrapper/gradle-wrapper.properties') }}
          restore-keys: |
            ${{ runner.os }}-gradle-

      - name: Compute CI version metadata
        run: |
          echo "CI_VERSION_CODE=$((GITHUB_RUN_NUMBER * 100 + GITHUB_RUN_ATTEMPT))" >> "$GITHUB_ENV"
          echo "CI_VERSION_NAME=0.1.0-ci.${GITHUB_RUN_NUMBER}.${GITHUB_RUN_ATTEMPT}-${GITHUB_SHA}" >> "$GITHUB_ENV"

      - name: Build Debug APK
        env:
          AVV_API_KEY: ${{ secrets.AVV_API_KEY }}
          DB_API_KEY: ${{ secrets.DB_API_KEY }}
          DB_CLIENT_ID: ${{ secrets.DB_CLIENT_ID }}
          DB_TIMETABLES_API_KEY: ${{ secrets.DB_TIMETABLES_API_KEY }}
          DB_TIMETABLES_CLIENT_ID: ${{ secrets.DB_TIMETABLES_CLIENT_ID }}
        run: cd catchit-android && ./gradlew assembleDebug --no-daemon --stacktrace

      - name: Upload Debug APK
        uses: actions/upload-artifact@v4
        with:
          name: CatchIt-debug-apk-${{ github.event_name }}-${{ github.run_number }}-${{ github.sha }}
          path: catchit-android/app/build/outputs/apk/debug/app-debug.apk
          retention-days: 30
